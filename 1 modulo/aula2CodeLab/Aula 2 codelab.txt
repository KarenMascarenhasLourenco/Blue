Exercício 5 - Elabore um programa que receba 3 notas de um aluno e calcule a média dele.
	let  nota1 = prompt("Digite a primeira nota ");
	let nota1Float=parseFloat(nota1);
	let  nota2 = prompt("Digite a segunda nota: ");
	let nota2Float =parseFloat(nota2);
	let nota3 = prompt("Digite a terceira nota: ");
	let nota3Float= parseFloat(nota3);
	console.log("A média é: "+ (nota1Float+nota2Float+nota3Float)/3);

Exemplo:
Primeira nota = 7
Segunda nota = 8
Terceira nota = 10

Média = 7,75
Exercício 6 - Elabore um programa que recebe dois valores inteiros e mostra se o primeiro valor é maior ou igual ao segundo valor

	let  valor1 = prompt("Digite a primeira nota ");
	let valor1Inteiro=parseInt(valor1);
	let  valor2 = prompt("Digite a segunda nota: ");
	let valor2Inteiro =parseInt(valor2);
	if (valor1Inteiro === valor2Inteiro) {
	console.log("O primeiro valor é igual ao segundo");
	}else if (valor1Inteiro >= valor2Inteiro)
	{
	console.log("O primeiro valor é maior que o segundo");
	}else{
	console.log("O primeiro valor é menor que o segundo")
	}

Exemplo:
